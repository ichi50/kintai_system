server {
    server_name kintai.mydns.jp;

    location / {
        include proxy_params;
        proxy_pass http://unix:/home/ubuntu/kintai_system/gunicorn.sock;
    }

    location /static/ {
        alias /home/ubuntu/kintai_system/static/;
    }

    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/kintai.mydns.jp/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/kintai.mydns.jp/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot

}
server {
    if ($host = kintai.mydns.jp) {
        return 301 https://$host$request_uri;
    } # managed by Certbot


    listen 80;
    server_name kintai.mydns.jp;
    return 404; # managed by Certbot


}
