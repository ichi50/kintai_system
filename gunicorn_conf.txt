[Unit]
Description=gunicorn daemon
After=network.target

[Service]
User=ubuntu
Group=www-data
WorkingDirectory=/home/ubuntu/kintai_system
ExecStart=/home/ubuntu/kintai_system/venv/bin/gunicorn \
          --access-logfile - \
          --workers 3 \
          --bind unix:/home/ubuntu/kintai_system/gunicorn.sock \
          kintai_system.wsgi:application
[Install]
WantedBy=multi-user.target
